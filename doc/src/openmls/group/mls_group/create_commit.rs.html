1//! Defines the `CreateCommit` trait and its implementation for `MlsGroup`.
2
3use super::*;
4use crate::{credentials::CredentialWithKey, treesync::LeafNodeParameters};
5
6/// Can be used to denote the type of a commit.
7#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
8pub(crate) enum CommitType {
9    External(CredentialWithKey),
10    Member,
11}
12
13pub(crate) struct CreateCommitParams<'a> {
14    framing_parameters: Option<FramingParameters<'a>>,
15    credential_with_key: CredentialWithKey,
16
17    inline_proposals: Vec<Proposal>,          // Optional
18    force_self_update: bool,                  // Optional
19    leaf_node_parameters: LeafNodeParameters, // Optional
20}
21
22pub(crate) struct TempBuilderCCPM0 {}
23
24pub(crate) struct CreateCommitParamsBuilder<'a> {
25    ccp: CreateCommitParams<'a>,
26}
27
28impl TempBuilderCCPM0 {
29    pub(crate) fn external_commit(
30        self,
31        credential_with_key: CredentialWithKey,
32        framing_parameters: FramingParameters,
33    ) -> CreateCommitParamsBuilder {
34        CreateCommitParamsBuilder {
35            ccp: CreateCommitParams {
36                framing_parameters: Some(framing_parameters),
37                credential_with_key,
38
39                // defaults:
40                inline_proposals: vec![],
41                force_self_update: true,
42                leaf_node_parameters: LeafNodeParameters::default(),
43            },
44        }
45    }
46}

48impl<'a> CreateCommitParamsBuilder<'a> {

51        self
52    }
53    pub(crate) fn build(self) -> CreateCommitParams<'a> {
54        self.ccp

79impl MlsGroup {
80    pub(crate) fn create_external_commit<Provider: OpenMlsProvider>(
81        &mut self,
82        params: CreateCommitParams,
83        provider: &Provider,
84        signer: &impl Signer,
85    ) -> Result<CreateCommitResult, CreateCommitError> {
86        // We  are building an external commit. This means we have to pull the
87        // framing parameters out of the create commit parameteres instead of the group. Since
88        // these are set together with the group mode, we can be sure that this is `Some(..)` (see
89        // [`TempBuilderCCPM0::external_commit`].
90        let framing_parameters = params.framing_parameters.unwrap();
91        let commit_type = CommitType::External(params.credential_with_key().clone());
92
93        let ciphersuite = self.ciphersuite();
94
95        let sender = match commit_type {
96            CommitType::External(_) => Sender::NewMemberCommit,
97            CommitType::Member => Sender::build_member(self.own_leaf_index()),
98        };
99
100        // Filter proposals
101        let (proposal_queue, contains_own_updates) = ProposalQueue::filter_proposals(
102            ciphersuite,
103            provider.crypto(),
104            sender.clone(),
105            self.proposal_store(),
106            params.inline_proposals(),
107            self.own_leaf_index(),
108        )
109        .map_err(|e| match e {
110            ProposalQueueError::LibraryError(e) => e.into(),
111            ProposalQueueError::ProposalNotFound => CreateCommitError::MissingProposal,
112            ProposalQueueError::UpdateFromExternalSender
113            | ProposalQueueError::SelfRemoveFromNonMember => {
114                CreateCommitError::WrongProposalSenderType
115            }
116        })?;
117
118        // 11.2:
119        // Proposals with a non-default proposal type MUST NOT be included in a commit
120        // unless the proposal type is supported by all the members of the group that
121        // will process the Commit (i.e., not including any members being added
122        // or removed by the Commit).
123
124        let proposal_reference_list = proposal_queue.commit_list();
125
126        // Validate the proposals by doing the following checks:
127
128        // ValSem113: All Proposals: The proposal type must be supported by all
129        // members of the group
130        self.public_group
131            .validate_proposal_type_support(&proposal_queue)?;
132        // ValSem101
133        // ValSem102
134        // ValSem103
135        // ValSem104
136        self.public_group
137            .validate_key_uniqueness(&proposal_queue, None)?;
138        // ValSem105
139        self.public_group.validate_add_proposals(&proposal_queue)?;
140        // ValSem106
141        // ValSem109
142        self.public_group.validate_capabilities(&proposal_queue)?;
143        // ValSem107
144        // ValSem108
145        self.public_group
146            .validate_remove_proposals(&proposal_queue)?;
147        self.public_group
148            .validate_pre_shared_key_proposals(&proposal_queue)?;
149        // Validate update proposals for member commits
150        if let Sender::Member(sender_index) = &sender {
151            // ValSem110
152            // ValSem111
153            // ValSem112
154            self.public_group
155                .validate_update_proposals(&proposal_queue, *sender_index)?;
156        }
157
158        // ValSem208
159        // ValSem209
160        self.public_group
161            .validate_group_context_extensions_proposal(&proposal_queue)?;
162
163        // Make a copy of the public group to apply proposals safely
164        let mut diff = self.public_group.empty_diff();
165
166        // Apply proposals to tree
167        let apply_proposals_values =
168            diff.apply_proposals(&proposal_queue, self.own_leaf_index())?;
169
170        let path_computation_result =
171            // If path is needed, compute path values
172            if apply_proposals_values.path_required
173                || contains_own_updates
174                || params.force_self_update()
175                || !params.leaf_node_parameters().is_empty()
176            {
177                // Process the path. This includes updating the provisional
178                // group context by updating the epoch and computing the new
179                // tree hash.
180                diff.compute_path(
181                    provider.rand(),
182                    provider.crypto(),
183                    self.own_leaf_index(),
184                    apply_proposals_values.exclusion_list(),
185                    &commit_type,
186                    params.leaf_node_parameters(),
187                    signer,
188                    apply_proposals_values.extensions.clone()
189                )?
190            } else {
191                // If path is not needed, update the group context and return
192                // empty path processing results
193                diff.update_group_context(provider.crypto(), apply_proposals_values.extensions.clone())?;
194                PathComputationResult::default()
195            };
196
197        let update_path_leaf_node = path_computation_result
198            .encrypted_path
199            .as_ref()
200            .map(|path| path.leaf_node().clone());
201
202        // Create commit message
203        let commit = Commit {
204            proposals: proposal_reference_list,
205            path: path_computation_result.encrypted_path,
206        };
207
208        // Build AuthenticatedContent
209        let mut authenticated_content = AuthenticatedContent::commit(
210            framing_parameters,
211            sender,
212            commit,
213            self.public_group.group_context(),
214            signer,
215        )?;
216
217        // Update the confirmed transcript hash using the commit we just created.
218        diff.update_confirmed_transcript_hash(provider.crypto(), &authenticated_content)?;
219
220        let serialized_provisional_group_context = diff
221            .group_context()
222            .tls_serialize_detached()
223            .map_err(LibraryError::missing_bound_check)?;
224
225        let joiner_secret = JoinerSecret::new(
226            provider.crypto(),
227            ciphersuite,
228            path_computation_result.commit_secret,
229            self.group_epoch_secrets().init_secret(),
230            &serialized_provisional_group_context,
231        )
232        .map_err(LibraryError::unexpected_crypto_error)?;
233
234        // Prepare the PskSecret
235        let psk_secret = {
236            let psks = load_psks(
237                provider.storage(),
238                &self.resumption_psk_store,
239                &apply_proposals_values.presharedkeys,
240            )?;
241
242            PskSecret::new(provider.crypto(), ciphersuite, psks)?
243        };
244
245        // Create key schedule
246        let mut key_schedule =
247            KeySchedule::init(ciphersuite, provider.crypto(), &joiner_secret, psk_secret)?;
248
249        let serialized_provisional_group_context = diff
250            .group_context()
251            .tls_serialize_detached()
252            .map_err(LibraryError::missing_bound_check)?;
253
254        let welcome_secret = key_schedule
255            .welcome(provider.crypto(), self.ciphersuite())
256            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
257        key_schedule
258            .add_context(provider.crypto(), &serialized_provisional_group_context)
259            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
260        let provisional_epoch_secrets = key_schedule
261            .epoch_secrets(provider.crypto(), self.ciphersuite())
262            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
263
264        // Calculate the confirmation tag
265        let confirmation_tag = provisional_epoch_secrets
266            .confirmation_key()
267            .tag(
268                provider.crypto(),
269                self.ciphersuite(),
270                diff.group_context().confirmed_transcript_hash(),
271            )
272            .map_err(LibraryError::unexpected_crypto_error)?;
273
274        // Set the confirmation tag
275        authenticated_content.set_confirmation_tag(confirmation_tag.clone());
276
277        diff.update_interim_transcript_hash(
278            ciphersuite,
279            provider.crypto(),
280            confirmation_tag.clone(),
281        )?;
282
283        // only computes the group info if necessary
284        let group_info = if !apply_proposals_values.invitation_list.is_empty()
285            || self.configuration().use_ratchet_tree_extension
286        {
287            // Create the ratchet tree extension if necessary
288            let external_pub = provisional_epoch_secrets
289                .external_secret()
290                .derive_external_keypair(provider.crypto(), ciphersuite)
291                .map_err(LibraryError::unexpected_crypto_error)?
292                .public;
293            let external_pub_extension =
294                Extension::ExternalPub(ExternalPubExtension::new(external_pub.into()));
295            let other_extensions: Extensions = if self.configuration().use_ratchet_tree_extension {
296                Extensions::from_vec(vec![
297                    Extension::RatchetTree(RatchetTreeExtension::new(diff.export_ratchet_tree())),
298                    external_pub_extension,
299                ])?
300            } else {
301                Extensions::single(external_pub_extension)
302            };
303
304            // Create to-be-signed group info.
305            let group_info_tbs = {
306                GroupInfoTBS::new(
307                    diff.group_context().clone(),
308                    other_extensions,
309                    confirmation_tag,
310                    self.own_leaf_index(),
311                )
312            };
313            // Sign to-be-signed group info.
314            Some(group_info_tbs.sign(signer)?)
315        } else {
316            None
317        };
318
319        // Check if new members were added and, if so, create welcome messages
320        let welcome_option = if !apply_proposals_values.invitation_list.is_empty() {
321            // Encrypt GroupInfo object
322            let (welcome_key, welcome_nonce) = welcome_secret
323                .derive_welcome_key_nonce(provider.crypto(), self.ciphersuite())
324                .map_err(LibraryError::unexpected_crypto_error)?;
325            let encrypted_group_info = welcome_key
326                .aead_seal(
327                    provider.crypto(),
328                    group_info
329                        .as_ref()
330                        .ok_or_else(|| LibraryError::custom("GroupInfo was not computed"))?
331                        .tls_serialize_detached()
332                        .map_err(LibraryError::missing_bound_check)?
333                        .as_slice(),
334                    &[],
335                    &welcome_nonce,
336                )
337                .map_err(LibraryError::unexpected_crypto_error)?;
338
339            // Create group secrets for later use, so we can afterwards consume the
340            // `joiner_secret`.
341            let encrypted_secrets = diff.encrypt_group_secrets(
342                &joiner_secret,
343                apply_proposals_values.invitation_list,
344                path_computation_result.plain_path.as_deref(),
345                &apply_proposals_values.presharedkeys,
346                &encrypted_group_info,
347                provider.crypto(),
348                self.own_leaf_index(),
349            )?;
350
351            // Create welcome message
352            let welcome = Welcome::new(self.ciphersuite(), encrypted_secrets, encrypted_group_info);
353            Some(welcome)
354        } else {
355            None
356        };
357
358        let (provisional_group_epoch_secrets, provisional_message_secrets) =
359            provisional_epoch_secrets.split_secrets(
360                serialized_provisional_group_context,
361                diff.tree_size(),
362                self.own_leaf_index(),
363            );
364
365        let staged_commit_state = MemberStagedCommitState::new(
366            provisional_group_epoch_secrets,
367            provisional_message_secrets,
368            diff.into_staged_diff(provider.crypto(), ciphersuite)?,
369            path_computation_result.new_keypairs,
370            // The committer is not allowed to include their own update
371            // proposal, so there is no extra keypair to store here.
372            None,
373            update_path_leaf_node,
374        );
375        let staged_commit = StagedCommit::new(
376            proposal_queue,
377            StagedCommitState::GroupMember(Box::new(staged_commit_state)),
378        );
379
380        Ok(CreateCommitResult {
381            commit: authenticated_content,
382            welcome_option,
383            staged_commit,
384            group_info: group_info.filter(|_| self.configuration().use_ratchet_tree_extension),
385        })
386    }
