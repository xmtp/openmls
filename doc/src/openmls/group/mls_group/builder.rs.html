54    /// Build a new group with the given group ID.
55    ///
56    /// If an [`MlsGroupCreateConfig`] is provided, it will be used to configure the
57    /// group. Otherwise, the internal builder is used to build one with the
58    /// parameters set on this builder.
59    pub(super) fn build_internal<Provider: OpenMlsProvider>(
60        self,
61        provider: &Provider,
62        signer: &impl Signer,
63        credential_with_key: CredentialWithKey,
64        mls_group_create_config_option: Option<MlsGroupCreateConfig>,
65    ) -> Result<MlsGroup, NewGroupError<Provider::StorageError>> {
66        let mls_group_create_config = mls_group_create_config_option
67            .unwrap_or_else(|| self.mls_group_create_config_builder.build());
68        let group_id = self
69            .group_id
70            .unwrap_or_else(|| GroupId::random(provider.rand()));
71        let ciphersuite = mls_group_create_config.ciphersuite;
72
73        let (public_group_builder, commit_secret, leaf_keypair) =
74            PublicGroup::builder(group_id, ciphersuite, credential_with_key)
75                .with_group_context_extensions(
76                    mls_group_create_config.group_context_extensions.clone(),
77                )?
78                .with_leaf_node_extensions(mls_group_create_config.leaf_node_extensions.clone())?
79                .with_lifetime(*mls_group_create_config.lifetime())
80                .with_capabilities(mls_group_create_config.capabilities.clone())
81                .get_secrets(provider, signer)
82                .map_err(|e| match e {
83                    PublicGroupBuildError::LibraryError(e) => NewGroupError::LibraryError(e),
84                    PublicGroupBuildError::InvalidExtensions(e) => e.into(),
85                })?;
86
87        let serialized_group_context = public_group_builder
88            .group_context()
89            .tls_serialize_detached()
90            .map_err(LibraryError::missing_bound_check)?;
91
92        // Derive an initial joiner secret based on the commit secret.
93        // Derive an epoch secret from the joiner secret.
94        // We use a random `InitSecret` for initialization.
95        let joiner_secret = JoinerSecret::new(
96            provider.crypto(),
97            ciphersuite,
98            commit_secret,
99            &InitSecret::random(ciphersuite, provider.rand())
100                .map_err(LibraryError::unexpected_crypto_error)?,
101            &serialized_group_context,
102        )
103        .map_err(LibraryError::unexpected_crypto_error)?;
104
105        // TODO(#1357)
106        let mut resumption_psk_store = ResumptionPskStore::new(32);
107
108        // Prepare the PskSecret
109        let psk_secret = load_psks(provider.storage(), &resumption_psk_store, &self.psk_ids)
110            .and_then(|psks| PskSecret::new(provider.crypto(), ciphersuite, psks))
111            .map_err(|e| {
112                log::debug!("Unexpected PSK error: {:?}", e);
113                LibraryError::custom("Unexpected PSK error")
114            })?;
115
116        let mut key_schedule =
117            KeySchedule::init(ciphersuite, provider.crypto(), &joiner_secret, psk_secret)?;
118        key_schedule
119            .add_context(provider.crypto(), &serialized_group_context)
120            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
121
122        let epoch_secrets = key_schedule
123            .epoch_secrets(provider.crypto(), ciphersuite)
124            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
125
126        let (group_epoch_secrets, message_secrets) = epoch_secrets.split_secrets(
127            serialized_group_context,
128            TreeSize::new(1),
129            LeafNodeIndex::new(0u32),
130        );
131
132        let initial_confirmation_tag = message_secrets
133            .confirmation_key()
134            .tag(provider.crypto(), ciphersuite, &[])
135            .map_err(LibraryError::unexpected_crypto_error)?;
136
137        let message_secrets_store = MessageSecretsStore::new_with_secret(
138            mls_group_create_config.max_past_epochs(),
139            message_secrets,
140        );
141
142        let public_group = public_group_builder
143            .with_confirmation_tag(initial_confirmation_tag)
144            .build(provider.crypto())?;
145
146        // We already add a resumption PSK for epoch 0 to make things more unified.
147        let resumption_psk = group_epoch_secrets.resumption_psk();
148        resumption_psk_store.add(public_group.group_context().epoch(), resumption_psk.clone());
149
150        let mls_group = MlsGroup {
151            mls_group_config: mls_group_create_config.join_config.clone(),
152            own_leaf_nodes: vec![],
153            aad: vec![],
154            group_state: MlsGroupState::Operational,
155            public_group,
156            group_epoch_secrets,
157            own_leaf_index: LeafNodeIndex::new(0),
158            message_secrets_store,
159            resumption_psk_store,
160        };
161
162        mls_group
163            .store(provider.storage())
164            .map_err(NewGroupError::StorageError)?;
165        mls_group
166            .store_epoch_keypairs(provider.storage(), &[leaf_keypair])
167            .map_err(NewGroupError::StorageError)?;
168
169        Ok(mls_group)
170    }

190    /// Sets the `max_past_epochs` property of the MlsGroup.
191    /// This allows application messages from previous epochs to be decrypted.
192    ///
193    /// **WARNING**
194    ///
195    /// This feature enables the storage of message secrets from past epochs.
196    /// It is a trade-off between functionality and forward secrecy and should only be enabled
197    /// if the Delivery Service cannot guarantee that application messages will be sent in
198    /// the same epoch in which they were generated. The number for `max_epochs` should be
199    /// as low as possible.
200    pub fn max_past_epochs(mut self, max_past_epochs: usize) -> Self {
201        self.mls_group_create_config_builder = self

205    }
